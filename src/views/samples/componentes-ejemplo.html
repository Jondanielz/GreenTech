🔧 Uso del ComponentsLoader:
import ComponentsLoader from './utils/components-loader.js';

// Inicializar componentes
ComponentsLoader.init();

// O cargar componentes dinámicamente
ComponentsLoader.loadCommonComponents({
  navbar: true,
  sidebar: true,
  footer: true,
  basePath: '../../components/'
});
